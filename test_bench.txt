library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity timer_tb is
end timer_tb;

architecture Behavioral of timer_tb is


component timer
Port (
clk : in STD_LOGIC;
reset : in STD_LOGIC;
enable : in STD_LOGIC;done : out STD_LOGIC
);
end component;


signal clk : std_logic := '0';
signal reset : std_logic := '0';
signal enable : std_logic := '0';
signal done_tb : std_logic;
signal p1_switch : std_logic := '0'; 
signal p2_switch : std_logic := '0'; 
signal p3_switch : std_logic := '0'; 
signal p4_switch : std_logic := '0'; 

begin



uut: entity work.timer
port map (
clk => clk,
reset => reset,
enable => enable,
done => done_tb,
p1_switch => p1_switch,
p2_switch => p2_switch,
p3_switch => p3_switch,
p4_switch => p4_switch
);



clk_process : process
begin
clk <= '0';
wait for 5 ns;
clk <= '1';
wait for 5 ns;
end process;


end Behavioral;
